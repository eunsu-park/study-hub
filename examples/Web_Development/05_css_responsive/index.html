8. Media Query Example Code

/* Mobile First approach */

/* Base (mobile) styles */
.container {
    padding: 10px;
}

/* Tablet (576px and above) */
@media (min-width: 576px) {
    .container {
        padding: 20px;
    }
}

/* Small desktop (768px and above) */
@media (min-width: 768px) {
    .container {
        max-width: 720px;
        margin: 0 auto;
    }
}

/* Desktop (992px and above) */
@media (min-width: 992px) {
    .container {
        max-width: 960px;
    }
}

/* Large screen (1200px and above) */
@media (min-width: 1200px) {
    .container {
        max-width: 1140px;
    }
}
