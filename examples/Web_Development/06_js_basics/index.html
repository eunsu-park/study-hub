JavaScript Basics Example

Output Results

1. Variables (var, let, const)

// var - function scope (not recommended)
var oldVar = "I'm var";

// let - block scope, reassignable
let count = 0;
count = 1;

// const - block scope, not reassignable
const PI = 3.14159;
const person = { name: "Hong Gildong" };
person.name = "Kim Cheolsu"; // Object properties can be changed

2. Data Types

// Primitive types
let str = "string";          // String
let num = 42;                // Number
let float = 3.14;            // Number (no distinction from integers)
let bool = true;             // Boolean
let empty = null;            // Null
let notDefined;              // Undefined
let sym = Symbol("id");      // Symbol
let big = 9007199254740991n; // BigInt

// Reference types
let obj = { key: "value" };  // Object
let arr = [1, 2, 3];         // Array
let func = function() {};    // Function

3. Operators

// Arithmetic operators
let a = 10, b = 3;
// +, -, *, /, %, **

// Comparison operators
// ==, ===, !=, !==, <, >, <=, >=
// == vs === (type comparison difference)

// Logical operators
// && (AND), || (OR), ! (NOT)

// Ternary operator
let result = a > b ? "greater" : "less";

// Nullish coalescing (??)
let value = null ?? "default value";

// Optional chaining (?.)
let obj = { nested: { value: 42 } };
let safe = obj?.nested?.value;

4. Conditionals

// if-else
if (condition1) {
    // execute
} else if (condition2) {
    // execute
} else {
    // execute
}

// switch
switch (value) {
    case 'A':
        break;
    case 'B':
        break;
    default:
        break;
}

5. Loops

// for
for (let i = 0; i < 5; i++) { }

// while
while (condition) { }

// do-while
do { } while (condition);

// for...of (array iteration)
for (const item of array) { }

// for...in (object key iteration)
for (const key in object) { }

// forEach, map, filter, reduce

6. Functions

// Function declaration
function greet(name) {
    return `Hello, ${name}!`;
}

// Function expression
const greet = function(name) {
    return `Hello, ${name}!`;
};

// Arrow function
const greet = (name) => `Hello, ${name}!`;

// Default parameters
const greet = (name = "Guest") => `Hello, ${name}!`;

// Rest parameters
const sum = (...numbers) => numbers.reduce((a, b) => a + b, 0);

// Callback function
setTimeout(() => console.log("After 1 second"), 1000);

7. Arrays

const fruits = ["Apple", "Banana", "Orange"];

// Basic methods
fruits.push("Grape");       // Add to end
fruits.pop();               // Remove from end
fruits.unshift("Strawberry"); // Add to beginning
fruits.shift();             // Remove from beginning

// Higher-order functions
const numbers = [1, 2, 3, 4, 5];
numbers.map(n => n * 2);           // [2, 4, 6, 8, 10]
numbers.filter(n => n > 2);        // [3, 4, 5]
numbers.find(n => n > 2);          // 3
numbers.reduce((acc, n) => acc + n, 0); // 15
numbers.every(n => n > 0);         // true
numbers.some(n => n > 4);          // true

8. Objects

// Object literal
const person = {
    name: "Hong Gildong",
    age: 25,
    greet() {
        return `Hello, I'm ${this.name}.`;
    }
};

// Property access
person.name;      // Dot notation
person["age"];    // Bracket notation

// Destructuring assignment
const { name, age } = person;

// Spread operator
const newPerson = { ...person, city: "Seoul" };

// Object methods
Object.keys(person);    // ["name", "age", "greet"]
Object.values(person);  // ["Hong Gildong", 25, ...]
Object.entries(person); // [["name", "Hong Gildong"], ...]

9. Classes (ES6)

class Animal {
    constructor(name) {
        this.name = name;
    }

    speak() {
        return `${this.name} makes a sound.`;
    }
}

class Dog extends Animal {
    constructor(name, breed) {
        super(name);
        this.breed = breed;
    }

    speak() {
        return `${this.name} barks!`;
    }
}

const dog = new Dog("Buddy", "Jindo");
