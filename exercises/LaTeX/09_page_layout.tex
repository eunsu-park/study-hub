% === Exercise 2: Headers and Footers ===
% Problem: Two-sided, page numbers on outer corners, section in inner header,
% title in outer header, author in footer center, date on outer corners,
% plain first page.

\section*{Exercise 2: Headers and Footers}

\pagestyle{fancy}
\fancyhf{}
\fancyhead[LE]{Page Layout Exercises}     % Even left = outer header
\fancyhead[RO]{Page Layout Exercises}     % Odd right = outer header
\fancyhead[RE]{\leftmark}                 % Even right = inner header (section)
\fancyhead[LO]{\leftmark}                 % Odd left = inner header (section)
\fancyfoot[C]{Study Project}              % Center footer: author
\fancyfoot[LE,RO]{\thepage}              % Outer corners: page number
\renewcommand{\headrulewidth}{0.4pt}
\renewcommand{\footrulewidth}{0.4pt}

The headers and footers are now configured.  On even (left) pages the title
appears on the left, and on odd (right) pages it appears on the right.
The section name appears on the inner side.

\begin{verbatim}
\pagestyle{fancy}
\fancyhf{}
\fancyhead[LE]{Document Title}
\fancyhead[RO]{Document Title}
\fancyhead[RE]{\leftmark}
\fancyhead[LO]{\leftmark}
\fancyfoot[C]{Author Name}
\fancyfoot[LE,RO]{\thepage}
\end{verbatim}

% === Exercise 7: Page Style Switching ===
% Problem: Three custom page styles, switching between them.

\section*{Exercise 7: Page Style Switching}

\begin{verbatim}
\documentclass{report}
\usepackage{fancyhdr}

% Style 1: Minimal
\fancypagestyle{minimal}{
  \fancyhf{}
  \fancyfoot[C]{\thepage}
  \renewcommand{\headrulewidth}{0pt}
}

% Style 2: Standard
\fancypagestyle{standard}{
  \fancyhf{}
  \fancyhead[L]{\leftmark}
  \fancyhead[R]{\thepage}
  \renewcommand{\headrulewidth}{0.4pt}
}

% Style 3: Detailed
\fancypagestyle{detailed}{
  \fancyhf{}
  \fancyhead[L]{Report Title}
  \fancyhead[C]{\leftmark}
  \fancyhead[R]{\rightmark}
  \fancyfoot[L]{Draft v1.0}
  \fancyfoot[C]{\thepage}
  \fancyfoot[R]{\today}
  \renewcommand{\headrulewidth}{0.4pt}
  \renewcommand{\footrulewidth}{0.4pt}
}

\begin{document}
\pagestyle{minimal}
\chapter{Preface}
Minimal style here ...

\pagestyle{standard}
\chapter{Introduction}
Standard style here ...

\pagestyle{detailed}
\chapter{Methods}
Detailed style here ...
\end{document}
\end{verbatim}

\section*{Exercise 8: Layout Troubleshooting}

\textbf{Issues in the original code:}
\begin{enumerate}
  \item \verb|margin=0.5in| -- too narrow for readability
  \item Missing \verb|\fancyhf{}| -- old headers/footers not cleared
  \item Header text too long -- will overflow
  \item \verb|\parindent{2in}| -- excessive indentation
  \item \verb|\linespread{3}| -- excessive line spacing
\end{enumerate}

\textbf{Fixed version:}

\begin{verbatim}
\documentclass{article}
\usepackage[margin=1in]{geometry}       % Fix: reasonable margins
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}                              % Fix: clear defaults
\fancyhead[L]{Report Title}            % Fix: shorter header
\fancyhead[R]{\thepage}
\renewcommand{\headrulewidth}{0.4pt}
\setlength{\parindent}{1.5em}          % Fix: standard indent
\linespread{1.2}                        % Fix: reasonable spacing

\begin{document}
\section{Introduction}
This is a properly formatted document with reasonable margins,
clear headers, standard indentation, and comfortable line spacing.
\end{document}
\end{verbatim}
